\de{
Sei $\varphi$ die Eulersche Phi-Funktion. Beweise, dass für alle natürlichen Zahlen $n$ gilt:
\[
2^{n(n+1)}\mid 32\cdot\varphi\left(2^{2^n}-1\right).
\]
}
\fr{Soit $\varphi$ la fonction phi d'Euler. Montrer que pour tous les nombres entiers strictement positifs $n$
\[
2^{n(n+1)}\mid 32\cdot\varphi\left(2^{2^n}-1\right).
\]
}
\ita{}

\en{
Let $\varphi$ denote the Euler phi-function. Prove that for every positive integer $n$
\[
2^{n(n+1)}\mid 32\cdot\varphi\left(2^{2^n}-1\right).
\]
}
