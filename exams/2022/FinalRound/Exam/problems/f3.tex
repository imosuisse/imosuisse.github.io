\en{Let $\N$ be the set of positive integers. Find all functions $f\colon \N \to \N$ such that both
\begin{itemize}
    \item $f(f(m)f(n)) = mn$
    \item $f(2022a+1) = 2022a+1$
\end{itemize}
hold for all positive integers $m,n$ and $a$.}

\de{Sei $\N$ die Menge der positiven ganzen Zahlen. Finde alle Funktionen $f\colon \N \to \N$, so dass die beiden Gleichungen
\begin{itemize}
    \item $f(f(m)f(n)) = mn$
    \item $f(2022a+1) = 2022a+1$
\end{itemize}
für alle positiven ganzen Zahlen $m,n$ und $a$ erfüllt sind.}

\fr{Soit $\N$ l'ensemble des nombres entiers strictement positifs. Trouver toutes les fonctions $f\colon\N\to\N$ telles que les deux équations
\begin{itemize}
    \item $f(f(m)f(n)) = mn$
    \item $f(2022a+1) = 2022a+1$
\end{itemize}
sont vérifiées pour tous les nombres entiers strictement positifs $m, n$ et $a$.}
