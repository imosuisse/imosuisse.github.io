<script>
  function selectYear(event, year) {
    all = document.getElementsByClassName("results");
    for (i = 0; i < all.length; i++) {
      if (all[i].classList.contains(year)) {
        all[i].style.display = "block";
      } else {
        all[i].style.display = "none";
      }
    }
  }
</script>
